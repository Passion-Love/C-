                类的静态成员  
//静态数据成员  
```
#include <iostream>
using namespace std;
class point{
    public:
    point(int xx,int yy):x(xx),y(yy){count++;}
    point(point &p):x(p.x),y(p.y){count++;}
    ~point(){count--;}
    void showcount(){
        cout<<count<<endl;
    }
    void show(){
        cout<<x<<","<<y<<endl;
    }
    private:
    int x,y;
    static int count;
};
int point::count=0;

int main(){
    point po(1,2);
    po.show();
    po.showcount();
    point po2(po);
    po2.show();
    po2.showcount();
}
```

//静态函数成员  
```
class A{
    public:
    static void fun(A a);
    private:
    int x;
};

void A::fun(A a){
    cout<<x;//错误，不能引用非静态数据
    cout<<a.x;//正确
}
```
