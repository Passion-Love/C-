```
//suboop.h
#ifndef _SubOOp_h_
#define _SubOOp_h_
class ThisFileClass{
private:
	int Stu_Age;
public:
	void SetAge(int age);
	int GetAge(void);
};
 
#endif



//suboop.cpp
#include "SubOOp.h"
#include <iostream>
void ThisFileClass::SetAge(int age)
{
	Stu_Age = age;/*应为加了::（域作用符），所以相当于这里的函数依然写在类的内部*/
}
int ThisFileClass::GetAge(void)
{
	int age = Stu_Age;
	std::cout << "student's name :" << Stu_Age << std::endl;
	return age;
}



//main.cpp
#include <iostream>
#include "SubOOp.h"
using namespace std;
int main()
{
	ThisFileClass Obj;
	Obj.SetAge(100);
	Obj.GetAge();
 
	system("pause");
	return 0;
}
```
